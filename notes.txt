Pace Calculator:
function paceCalculator(distance, timeInMinutes) {
    let pacePerMile = timeInMinutes / distance
    console.log("Total distance: " + distance + " miles")
    console.log("Finish time: " + timeInMinutes + " minutes")
    console.log("Pace: " + pacePerMile + " minutes per mile")
    if (pacePerMile < 10) {
        console.log("Speedy!")
    } else {
        console.log("Solid effort! Keep training")
    }
}
